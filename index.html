async function generate() { const apiKey = "YOUR_GEMINI_API_KEY"; const prompt = document.getElementById("prompt").value; const response = await fetch( `https://generativelanguage.googleapis.com/v1beta/models/gemini-pro:generateContent?key=${apiKey}`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }], }), } ); const data = await response.json(); document.getElementById("output").innerText = data?.candidates?.[0]?.content?.parts?.[0]?.text || "No output."; }
